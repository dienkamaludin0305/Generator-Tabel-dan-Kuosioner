import React, { useState, useEffect, useRef } from 'react';
import { 
  Users, User, MessageSquare, Play, RefreshCcw, 
  AlertCircle, Sparkles, Gavel, Mic, MicOff,
  Settings, BookOpen, Send, ShieldAlert, Brain,
  Activity, Lock, Unlock, UserCircle, 
  Upload, CheckCircle2, Target, ShieldQuestion,
  ChevronRight, Volume2, Smile, Frown, Meh, Trophy,
  Star, Clock, Heart, Scale, ClipboardCheck,
  GraduationCap, School, MapPin, Zap
} from 'lucide-react';

export default function App() {
  const [step, setStep] = useState('setup'); 
  const [activeDosenIdx, setActiveDosenIdx] = useState(0);
  const [isProcessing, setIsProcessing] = useState(false);
  const [isReadingFile, setIsReadingFile] = useState(false);
  const [isContextLocked, setIsContextLocked] = useState(false);
  
  // State Recording
  const [isRecording, setIsRecording] = useState(false);
  const recognitionRef = useRef(null);

  // State Data Akademis Lengkap
  const [skripsiData, setSkripsiData] = useState({
    nama: '',
    prodi: '',
    fakultas: '',
    jenjang: 'S1',
    tipeSidang: 'PROPOSAL',
    judul: '',
    abstrak: '',
  });

  const [dosenList] = useState([
    { id: 1, nama: 'Prof. Dr. Hardjono', pakar: 'Metodologi', karakter: 'Killer' },
    { id: 2, nama: 'Dr. Sarah M.Si', pakar: 'Teori Sosial', karakter: 'Normal' },
    { id: 3, nama: 'Bapak Aris, M.T', pakar: 'Implementasi', karakter: 'Friendly' },
  ]);

  const [chatHistory, setChatHistory] = useState([]);
  const [currentInput, setCurrentInput] = useState('');
  const [dosenSatisfaction, setDosenSatisfaction] = useState(0);
  const [startTime, setStartTime] = useState(null);
  
  // Metrics Performa
  const [performanceLog, setPerformanceLog] = useState([]);
  const chatEndRef = useRef(null);

  // Fungsi Simulasi Otomatis (Demo)
  const runAutoSimulasi = () => {
    setSkripsiData({
      nama: 'Budi Darmawan',
      prodi: 'Teknik Informatika',
      fakultas: 'Teknik',
      jenjang: 'S1',
      tipeSidang: 'SKRIPSI',
      judul: 'Penerapan Deep Learning untuk Deteksi Hama Padi Berbasis Mobile',
      abstrak: 'Penelitian ini bertujuan mengimplementasikan algoritma CNN untuk membantu petani mendeteksi hama secara real-time melalui smartphone.'
    });
    setIsContextLocked(true);
  };

  useEffect(() => {
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (SpeechRecognition) {
      recognitionRef.current = new SpeechRecognition();
      recognitionRef.current.continuous = false;
      recognitionRef.current.lang = 'id-ID';
      recognitionRef.current.onresult = (event) => {
        const transcript = event.results[0][0].transcript;
        setCurrentInput(prev => prev + (prev ? ' ' : '') + transcript);
        setIsRecording(false);
      };
      recognitionRef.current.onend = () => setIsRecording(false);
    }
  }, []);

  const toggleRecording = () => {
    if (isRecording) {
      recognitionRef.current?.stop();
    } else {
      setIsRecording(true);
      recognitionRef.current?.start();
    }
  };

  const handleFileUpload = (e) => {
    const file = e.target.files[0];
    if (!file) return;
    setIsReadingFile(true);
    setTimeout(() => {
      setSkripsiData(prev => ({
        ...prev,
        judul: "Analisis Efektivitas Operasional sebagai Variabel Moderasi pada PT Indonesia Comnets Plus",
        abstrak: "Penelitian ini menggunakan metode kuantitatif dengan sampel 150 karyawan untuk melihat pengaruh efektivitas operasional terhadap kinerja perusahaan."
      }));
      setIsReadingFile(false);
    }, 1500);
  };

  const generateAiQuestion = (dosenIdx, isFirst = false, prevAnswer = "", responseTime = 0) => {
    setIsProcessing(true);
    const dosen = dosenList[dosenIdx];
    
    setTimeout(() => {
      let text = "";
      let satisfactionGain = 0;

      if (isFirst) {
        text = `Selamat pagi. Saya ${dosen.nama}. Saya akan menguji aspek ${dosen.pakar} untuk sidang ${skripsiData.tipeSidang.toLowerCase()} Anda. Bisa jelaskan mengapa topik '${skripsiData.judul.split(' ').slice(0,3).join(' ')}...' ini penting untuk diangkat?`;
      } else {
        const hasPoliteWords = /mohon izin|terima kasih|baik bapak|baik ibu/i.test(prevAnswer);
        const structureScore = prevAnswer.split(' ').length > 15 ? 30 : 15;
        const speedScore = responseTime < 5000 ? 20 : 10;
        
        satisfactionGain = structureScore + speedScore + (hasPoliteWords ? 15 : 0);
        
        setPerformanceLog(prev => [...prev, {
          dosenIdx,
          accuracy: prevAnswer.length > 40 ? 85 : 60,
          speed: responseTime,
          politeness: hasPoliteWords,
          structure: structureScore
        }]);

        const currentTotal = dosenSatisfaction + satisfactionGain;
        
        if (currentTotal >= 100) {
          handleAutoTransition();
          setIsProcessing(false);
          return;
        }

        setDosenSatisfaction(currentTotal);
        const followUps = [
          `Saya rasa penjelasan Anda belum mendalam di sisi ${dosen.pakar}. Bagaimana landasan teorinya?`,
          `Oke, tapi bagaimana validitas data Anda jika variabel ini ternyata tidak signifikan di lapangan?`,
          `Coba jelaskan kaitan teknisnya dengan masalah yang Anda angkat sekali lagi.`
        ];
        text = followUps[Math.floor(Math.random() * followUps.length)];
      }

      setChatHistory(prev => [...prev, {
        nama: dosen.nama,
        text,
        type: 'question',
        karakter: dosen.karakter
      }]);
      setStartTime(Date.now());
      setIsProcessing(false);
    }, 1200);
  };

  const handleAutoTransition = () => {
    const currentDosen = dosenList[activeDosenIdx];
    setChatHistory(prev => [...prev, { 
      type: 'system', 
      text: `[SISTEM] ${currentDosen.nama} merasa puas dengan jawaban Anda.` 
    }]);

    if (activeDosenIdx < dosenList.length - 1) {
      setTimeout(() => {
        const nextIdx = activeDosenIdx + 1;
        setActiveDosenIdx(nextIdx);
        setDosenSatisfaction(0);
        generateAiQuestion(nextIdx, true);
      }, 2000);
    } else {
      setTimeout(() => setStep('deliberation'), 2000);
    }
  };

  const handleSendMessage = () => {
    if (!currentInput.trim()) return;
    const answer = currentInput;
    const responseTime = Date.now() - startTime;
    setChatHistory(prev => [...prev, { text: answer, type: 'answer' }]);
    setCurrentInput('');
    generateAiQuestion(activeDosenIdx, false, answer, responseTime);
  };

  const calculateFinalScores = () => {
    const total = performanceLog.length || 1;
    const avgAccuracy = performanceLog.reduce((acc, curr) => acc + curr.accuracy, 0) / total;
    const avgSpeed = performanceLog.reduce((acc, curr) => acc + (curr.speed < 10000 ? 90 : 60), 0) / total;
    const avgPoliteness = (performanceLog.filter(p => p.politeness).length / total) * 100;
    const avgStructure = (performanceLog.reduce((acc, curr) => acc + curr.structure, 0) / (total * 30)) * 100;

    return {
      umum: Math.round((avgAccuracy * 0.4) + (avgStructure * 0.3) + (avgPoliteness * 0.3)),
      teknis: Math.round(avgAccuracy),
      mental: Math.round((avgPoliteness * 0.5) + (avgSpeed * 0.5)),
      bahasa: Math.round(avgStructure)
    };
  };

  useEffect(() => {
    chatEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [chatHistory]);

  return (
    <div className="min-h-screen bg-[#F8FAFC] text-slate-900 font-sans">
      <header className="bg-white/90 backdrop-blur-md border-b px-8 py-4 flex justify-between items-center sticky top-0 z-50">
        <div className="flex items-center gap-3">
          <div className="p-2 bg-indigo-600 rounded-xl shadow-lg shadow-indigo-200">
            <Gavel className="text-white w-5 h-5" />
          </div>
          <h1 className="text-lg font-black tracking-tight">SIDANG<span className="text-indigo-600">MASTER</span></h1>
        </div>
      </header>

      <main className="max-w-7xl mx-auto p-6">
        {step === 'setup' && (
          <div className="max-w-4xl mx-auto space-y-8 animate-in fade-in slide-in-from-bottom-4 duration-700">
            {/* Tombol Simulasi Cepat */}
            <div className="flex justify-end">
              <button 
                onClick={runAutoSimulasi}
                className="flex items-center gap-2 px-4 py-2 bg-amber-50 text-amber-700 rounded-xl border border-amber-200 text-[10px] font-black uppercase tracking-widest hover:bg-amber-100 transition-all shadow-sm"
              >
                <Zap className="w-3 h-3 fill-current" /> Auto-Fill Data (Simulasi)
              </button>
            </div>

            <div className={`bg-white p-10 rounded-[3rem] shadow-sm border-2 transition-all ${isContextLocked ? 'border-emerald-500/30 ring-4 ring-emerald-50' : 'border-slate-100'}`}>
              <div className="flex items-center justify-between mb-10">
                <div className="flex items-center gap-4">
                  <div className="w-12 h-12 rounded-2xl bg-indigo-50 text-indigo-600 flex items-center justify-center shadow-inner">
                    <UserCircle className="w-6 h-6" />
                  </div>
                  <div>
                    <h2 className="text-2xl font-black tracking-tight">Data Mahasiswa</h2>
                    <p className="text-[10px] font-bold text-slate-400 uppercase tracking-[0.2em]">Informasi Akademik & Penelitian</p>
                  </div>
                </div>
                <button 
                  onClick={() => setIsContextLocked(!isContextLocked)} 
                  className={`px-6 py-2.5 rounded-2xl text-[10px] font-black uppercase tracking-widest transition-all flex items-center gap-2 ${isContextLocked ? 'bg-emerald-500 text-white shadow-lg shadow-emerald-200' : 'bg-slate-100 text-slate-600 hover:bg-slate-200'}`}
                >
                  {isContextLocked ? <Lock className="w-3 h-3" /> : <Unlock className="w-3 h-3" />}
                  {isContextLocked ? 'Konteks Terkunci' : 'Kunci Konteks'}
                </button>
              </div>

              <div className={`space-y-8 ${isContextLocked ? 'opacity-60 pointer-events-none' : ''}`}>
                <div className="space-y-3">
                  <label className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">Nama Lengkap Mahasiswa</label>
                  <div className="relative group">
                    <div className="absolute inset-y-0 left-0 pl-5 flex items-center pointer-events-none text-slate-300 group-focus-within:text-indigo-500 transition-colors">
                      <User className="w-5 h-5" />
                    </div>
                    <input 
                      className="w-full pl-14 pr-6 py-5 bg-slate-50 border-2 border-transparent focus:border-indigo-500 focus:bg-white rounded-[1.5rem] outline-none font-bold text-slate-700 transition-all shadow-inner" 
                      value={skripsiData.nama} 
                      onChange={e => setSkripsiData({...skripsiData, nama: e.target.value})} 
                      placeholder="Contoh: Ahmad Subarjo" 
                    />
                  </div>
                </div>

                <div className="grid grid-cols-1 md:grid-cols-12 gap-6">
                  <div className="md:col-span-5 space-y-3">
                    <label className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">Program Studi</label>
                    <input 
                      className="w-full px-6 py-5 bg-slate-50 border-2 border-transparent focus:border-indigo-500 focus:bg-white rounded-[1.5rem] outline-none font-bold text-slate-700 transition-all shadow-inner" 
                      value={skripsiData.prodi} 
                      onChange={e => setSkripsiData({...skripsiData, prodi: e.target.value})} 
                      placeholder="Contoh: Teknik Informatika" 
                    />
                  </div>
                  <div className="md:col-span-4 space-y-3">
                    <label className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">Fakultas</label>
                    <input 
                      className="w-full px-6 py-5 bg-slate-50 border-2 border-transparent focus:border-indigo-500 focus:bg-white rounded-[1.5rem] outline-none font-bold text-slate-700 transition-all shadow-inner" 
                      value={skripsiData.fakultas} 
                      onChange={e => setSkripsiData({...skripsiData, fakultas: e.target.value})} 
                      placeholder="Teknik" 
                    />
                  </div>
                  <div className="md:col-span-3 space-y-3">
                    <label className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">Jenjang</label>
                    <select 
                      className="w-full px-6 py-5 bg-slate-50 border-2 border-transparent focus:border-indigo-500 focus:bg-white rounded-[1.5rem] outline-none font-bold text-slate-700 transition-all shadow-inner appearance-none cursor-pointer"
                      value={skripsiData.jenjang}
                      onChange={e => setSkripsiData({...skripsiData, jenjang: e.target.value})}
                    >
                      <option value="S1">S1</option>
                      <option value="S2">S2</option>
                      <option value="D3">D3</option>
                    </select>
                  </div>
                </div>

                <div className="flex bg-slate-100 p-2 rounded-[1.5rem] gap-2">
                  {['PROPOSAL', 'HASIL', 'SKRIPSI'].map((t) => (
                    <button
                      key={t}
                      onClick={() => setSkripsiData({...skripsiData, tipeSidang: t})}
                      className={`flex-1 py-3.5 rounded-[1rem] text-[10px] font-black uppercase tracking-widest transition-all ${skripsiData.tipeSidang === t ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-400 hover:text-slate-600'}`}
                    >
                      {t}
                    </button>
                  ))}
                </div>

                <div className="space-y-6">
                  <div className="space-y-3">
                    <label className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">Judul Penelitian</label>
                    <input 
                      className="w-full px-6 py-5 bg-slate-50 border-2 border-transparent focus:border-indigo-500 focus:bg-white rounded-[1.5rem] outline-none font-bold text-slate-700 transition-all shadow-inner" 
                      value={skripsiData.judul} 
                      onChange={e => setSkripsiData({...skripsiData, judul: e.target.value})} 
                      placeholder="Masukkan judul lengkap di sini..." 
                    />
                  </div>

                  <div className="space-y-3">
                    <label className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">Abstrak / Ringkasan Penelitian</label>
                    <textarea 
                      className="w-full px-6 py-6 bg-slate-50 border-2 border-transparent focus:border-indigo-500 focus:bg-white rounded-[2rem] outline-none text-sm font-medium leading-relaxed min-h-[150px] transition-all shadow-inner" 
                      value={skripsiData.abstrak} 
                      onChange={e => setSkripsiData({...skripsiData, abstrak: e.target.value})} 
                      placeholder="Tempelkan abstrak di sini..." 
                    />
                  </div>
                </div>
              </div>
            </div>

            <button 
              onClick={() => { setStep('simulation'); generateAiQuestion(0, true); }}
              disabled={!isContextLocked || !skripsiData.nama || !skripsiData.judul}
              className="w-full py-6 bg-slate-900 text-white rounded-[2.5rem] font-black text-sm uppercase tracking-[0.3em] shadow-2xl hover:bg-indigo-600 active:scale-95 transition-all disabled:opacity-20 flex items-center justify-center gap-4"
            >
              <Play className="w-5 h-5 fill-current" /> Mulai Simulasi Sidang
            </button>
          </div>
        )}

        {step === 'simulation' && (
          <div className="grid grid-cols-1 lg:grid-cols-12 gap-8 h-[calc(100vh-160px)]">
            <div className="lg:col-span-3 space-y-4">
              <div className="bg-white p-6 rounded-[2rem] border border-slate-200">
                <h4 className="text-[10px] font-black text-slate-400 uppercase tracking-widest mb-6 flex items-center gap-2">
                   <Activity className="w-3 h-3" /> Analisis Live
                </h4>
                <div className="space-y-6">
                   <div className="space-y-2">
                      <div className="flex justify-between items-end">
                         <span className="text-[10px] font-black uppercase text-slate-500">Kepuasan Dosen</span>
                         <span className="text-xs font-black text-indigo-600">{dosenSatisfaction}%</span>
                      </div>
                      <div className="h-2 bg-slate-100 rounded-full overflow-hidden">
                         <div className="h-full bg-indigo-500 transition-all" style={{ width: `${dosenSatisfaction}%` }}></div>
                      </div>
                   </div>
                   <div className="p-4 bg-indigo-50 rounded-2xl border border-indigo-100">
                      <p className="text-[10px] font-black text-indigo-600 uppercase mb-2">Aspek Kepakaran</p>
                      <div className="flex items-center gap-2">
                         <div className="w-8 h-8 rounded-lg bg-indigo-600 flex items-center justify-center text-white font-black text-xs">
                            {dosenList[activeDosenIdx].pakar[0]}
                         </div>
                         <p className="text-xs font-bold text-indigo-900">{dosenList[activeDosenIdx].pakar}</p>
                      </div>
                   </div>
                </div>
              </div>
            </div>

            <div className="lg:col-span-6 bg-white rounded-[3rem] border border-slate-200 shadow-2xl flex flex-col overflow-hidden">
              <div className="p-6 border-b flex justify-between items-center bg-white">
                <div className="flex items-center gap-4">
                  <div className="w-12 h-12 rounded-2xl bg-slate-900 flex items-center justify-center text-white shadow-lg">
                    <User className="w-6 h-6" />
                  </div>
                  <div>
                    <h3 className="font-black text-slate-800">{dosenList[activeDosenIdx].nama}</h3>
                    <span className="text-[9px] font-black text-indigo-600 uppercase tracking-widest">Penguji {activeDosenIdx + 1} ({dosenList[activeDosenIdx].karakter})</span>
                  </div>
                </div>
              </div>

              <div className="flex-1 overflow-y-auto p-8 space-y-6 bg-slate-50/30">
                {chatHistory.map((msg, i) => (
                  <div key={i} className={`flex ${msg.type === 'answer' ? 'justify-end' : 'justify-start'}`}>
                    {msg.type === 'system' ? (
                      <div className="w-full py-2 flex justify-center text-[9px] font-black text-emerald-600 uppercase">
                        {msg.text}
                      </div>
                    ) : (
                      <div className={`max-w-[85%] p-5 text-sm leading-relaxed shadow-sm ${msg.type === 'answer' ? 'bg-indigo-600 text-white rounded-[2rem] rounded-tr-none' : 'bg-white text-slate-800 rounded-[2rem] rounded-tl-none border border-slate-100'}`}>
                        {msg.text}
                      </div>
                    )}
                  </div>
                ))}
                {isProcessing && (
                  <div className="flex justify-start">
                    <div className="bg-white px-4 py-3 rounded-2xl flex gap-1 shadow-sm border border-slate-100">
                      <div className="w-1 h-1 bg-indigo-400 rounded-full animate-bounce"></div>
                      <div className="w-1 h-1 bg-indigo-400 rounded-full animate-bounce delay-75"></div>
                    </div>
                  </div>
                )}
                <div ref={chatEndRef} />
              </div>

              <div className="p-6 bg-white border-t">
                <div className="bg-slate-50 rounded-[2.5rem] border-2 border-slate-200 p-3 flex items-end gap-3 transition-all shadow-inner">
                  <button onClick={toggleRecording} className={`w-14 h-14 rounded-[1.5rem] flex items-center justify-center transition-all ${isRecording ? 'bg-red-500 text-white animate-pulse' : 'bg-white text-slate-400 border border-slate-200'}`}>
                    {isRecording ? <MicOff className="w-6 h-6" /> : <Mic className="w-6 h-6" />}
                  </button>
                  <textarea 
                    className="flex-1 bg-transparent border-none outline-none text-sm p-4 resize-none h-[60px] font-bold text-slate-700"
                    placeholder="Tulis jawaban Anda..."
                    value={currentInput}
                    onChange={e => setCurrentInput(e.target.value)}
                  />
                  <button onClick={handleSendMessage} disabled={!currentInput.trim() || isProcessing} className="w-14 h-14 bg-slate-900 text-white rounded-[1.5rem] flex items-center justify-center hover:bg-indigo-600 transition-all disabled:opacity-20 shadow-xl">
                    <Send className="w-6 h-6" />
                  </button>
                </div>
              </div>
            </div>

            <div className="lg:col-span-3 space-y-4">
               <div className="bg-white p-6 rounded-[2rem] border border-slate-200 text-center">
                  <h4 className="text-[10px] font-black text-slate-400 uppercase tracking-widest mb-4">Tips Simulasi</h4>
                  <p className="text-[11px] text-slate-500 font-medium leading-relaxed italic">"Gunakan kata-kata sopan seperti 'Izin menjawab Bapak/Ibu' untuk meningkatkan kepuasan penguji."</p>
               </div>
            </div>
          </div>
        )}

        {step === 'deliberation' && (
          <div className="max-w-4xl mx-auto py-10 animate-in zoom-in-95">
             <div className="bg-white p-12 rounded-[4rem] shadow-2xl border border-slate-200 space-y-10 text-center">
                <Users className="w-12 h-12 mx-auto text-amber-500" />
                <h2 className="text-3xl font-black uppercase">Rapat Tim Penguji</h2>
                <p className="text-slate-500 font-bold uppercase text-[10px]">Menghitung skor akhir mahasiswa...</p>
                <button onClick={() => setStep('result')} className="w-full py-6 bg-indigo-600 text-white rounded-[2rem] font-black uppercase shadow-xl">
                   Lihat Hasil Sidang
                </button>
             </div>
          </div>
        )}

        {step === 'result' && (
          <div className="max-w-4xl mx-auto py-10">
             <div className="bg-white p-16 rounded-[4rem] shadow-2xl border border-slate-200 text-center space-y-8">
                <Trophy className="w-16 h-16 mx-auto text-amber-500" />
                <h2 className="text-4xl font-black text-slate-900 uppercase">HASIL SIDANG</h2>
                <div className="p-6 bg-slate-50 rounded-3xl">
                   <p className="text-[10px] font-black text-slate-400 uppercase">Nama Mahasiswa</p>
                   <p className="text-xl font-black text-indigo-600">{skripsiData.nama}</p>
                </div>
                <div className="grid grid-cols-2 gap-4">
                   {Object.entries(calculateFinalScores()).map(([k, v]) => (
                     <div key={k} className="p-4 bg-slate-50 rounded-2xl border border-slate-100">
                        <p className="text-[9px] font-black text-slate-400 uppercase">{k}</p>
                        <p className="text-2xl font-black">{v}</p>
                     </div>
                   ))}
                </div>
                <button onClick={() => window.location.reload()} className="w-full py-6 border-2 border-slate-200 rounded-[2rem] font-black uppercase text-[10px]">Ulangi Simulasi</button>
             </div>
          </div>
        )}
      </main>
    </div>
  );
}
